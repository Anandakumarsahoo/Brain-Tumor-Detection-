import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, roc_curve, auc, f1_score, recall_score, precision_score, confusion_matrix
import os
import cv2
import tensorflow as tf
from tensorflow.keras import layers, models, regularizers
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras.optimizers import Adam
import seaborn as sns  # For better visualization of the confusion matrix
import pywt  # For wavelet transforms

# Define tumor classes
path = os.listdir('brain_tumor/Training/')
classes = {'no_tumor': 0, 'pituitary_tumor': 1}

# Load and preprocess data
X = []
Y = []
for cls in classes:
    pth = 'brain_tumor/Training/' + cls
    for j in os.listdir(pth):
        # Load image in grayscale for CNN
        img = cv2.imread(pth + '/' + j, 0)
        img = cv2.resize(img, (200, 200))
        img_filtered = cv2.GaussianBlur(img, (5, 5), 0)  # Gaussian blur
        X.append(img_filtered)
        Y.append(classes[cls])

X = np.array(X)
Y = np.array(Y)

# Function to apply wavelet transform and reconstruct image
def apply_wavelet_transform(image, wavelet='haar', level=1):
    # Apply wavelet transform
    coeffs = pywt.wavedec2(image, wavelet, level=level)
    # Reconstruct image using only the approximation coefficients
    reconstructed_image = pywt.waverec2(coeffs, wavelet)
    return reconstructed_image

# Apply wavelet transform to all images
X_wavelet = []
for img in X:
    img_wavelet = apply_wavelet_transform(img, wavelet='haar', level=1)
    X_wavelet.append(img_wavelet)

X_wavelet = np.array(X_wavelet)

# Reshape and normalize data for CNN
X_wavelet_cnn = X_wavelet.reshape(len(X_wavelet), 200, 200, 1)  # Reshape for CNN input
X_wavelet_cnn = X_wavelet_cnn / 255.0  # Normalize pixel values
Y_cnn = tf.keras.utils.to_categorical(Y, num_classes=2)  # One-hot encode labels

# Split the dataset for CNN
xtrain_wavelet, xtest_wavelet, ytrain_wavelet, ytest_wavelet = train_test_split(X_wavelet_cnn, Y_cnn, random_state=10, test_size=.20)

# Data Augmentation
datagen = ImageDataGenerator(
    rotation_range=20,      # Randomly rotate images by 20 degrees
    width_shift_range=0.2,  # Randomly shift images horizontally by 20%
    height_shift_range=0.2, # Randomly shift images vertically by 20%
    shear_range=0.2,        # Apply shear transformations
    zoom_range=0.2,         # Randomly zoom in/out by 20%
    horizontal_flip=True,   # Randomly flip images horizontally
    fill_mode='nearest'     # Fill missing pixels with the nearest value
)

# Fit the data generator to the training data
datagen.fit(xtrain_wavelet)

# Build Enhanced CNN model with Regularization
def build_cnn_model(input_shape):
    model = models.Sequential([
        layers.Conv2D(32, (3, 3), activation='relu', input_shape=input_shape, kernel_regularizer=regularizers.l2(0.001)),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(64, (3, 3), activation='relu', kernel_regularizer=regularizers.l2(0.001)),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(128, (3, 3), activation='relu', kernel_regularizer=regularizers.l2(0.001)),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(256, (3, 3), activation='relu', kernel_regularizer=regularizers.l2(0.001)),
        layers.MaxPooling2D((2, 2)),
        layers.Flatten(),
        layers.Dense(512, activation='relu', kernel_regularizer=regularizers.l2(0.001)),  # Increased neurons
        layers.Dropout(0.5),
        layers.Dense(256, activation='relu', kernel_regularizer=regularizers.l2(0.001)),  # Increased neurons
        layers.Dropout(0.5),
        layers.Dense(2, activation='softmax')
    ])
    return model

# Initialize CNN model
input_shape = (200, 200, 1)  # Input shape for grayscale images
cnn_model = build_cnn_model(input_shape)

# Compile CNN model with a lower learning rate
optimizer = Adam(learning_rate=0.0001)  # Lower learning rate for fine-tuning
cnn_model.compile(optimizer=optimizer,
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])

# Train CNN model with data augmentation
history_wavelet = cnn_model.fit(datagen.flow(xtrain_wavelet, ytrain_wavelet, batch_size=32),
                                epochs=20,  # Increased epochs for better learning
                                validation_data=(xtest_wavelet, ytest_wavelet))

# Evaluate CNN model
wavelet_test_loss, wavelet_test_acc = cnn_model.evaluate(xtest_wavelet, ytest_wavelet)
print(f"Wavelet CNN Test Accuracy: {wavelet_test_acc:.4f}")

# Plot CNN training history
plt.figure(figsize=(8, 6))
plt.plot(history_wavelet.history['accuracy'], label='Training Accuracy')
plt.plot(history_wavelet.history['val_accuracy'], label='Validation Accuracy')
plt.xlabel('Epochs', fontsize=14)
plt.ylabel('Accuracy', fontsize=14)
plt.legend(fontsize=12)
plt.title('Wavelet CNN Training History', fontsize=16, pad=20)
plt.grid(True, alpha=0.3)
plt.xticks(fontsize=12)
plt.yticks(fontsize=12)
plt.show()

# ROC Curve for CNN
cnn_probs = cnn_model.predict(xtest_wavelet)[:, 1]
fpr_cnn, tpr_cnn, _ = roc_curve(ytest_wavelet.argmax(axis=1), cnn_probs)
roc_auc_cnn = auc(fpr_cnn, tpr_cnn)

# Plot ROC Curve for CNN
plt.figure(figsize=(8, 6))
plt.plot(fpr_cnn, tpr_cnn, color='blue', lw=2, label=f'CNN (AUC = {roc_auc_cnn:.2f})')
plt.plot([0, 1], [0, 1], color='gray', lw=1, linestyle='--')
plt.xlabel('False Positive Rate', fontsize=14)

plt.title('ROC Curve for Wavelet CNN Model', fontsize=16, pad=20)
plt.legend(loc='lower right', fontsize=12)

# Print the metrics

# Plot Confusion Matrix with Enhanced Formatting
plt.figure(figsize=(8, 6))
sns.heatmap(conf_matrix, annot=True, fmt='d', cmap='Blues', cbar=True,
            xticklabels=['No Tumor', 'Pituitary Tumor'],
            yticklabels=['No Tumor', 'Pituitary Tumor'],
            annot_kws={"size": 14},  # Increase annotation font size
            linewidths=0.5,  # Add lines between cells
            square=True)  # Ensure the heatmap is square
plt.xlabel('Predicted Labels', fontsize=14)
plt.ylabel('True Labels', fontsize=14)
plt.title('Confusion Matrix for Wavelet CNN Model', fontsize=16, pad=20)
plt.xticks(fontsize=12)
plt.yticks(fontsize=12)
plt.show()
